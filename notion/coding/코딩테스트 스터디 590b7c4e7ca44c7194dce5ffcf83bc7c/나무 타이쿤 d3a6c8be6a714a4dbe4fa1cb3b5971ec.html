    #include <iostream>
#include <vector>

using namespace std;

int n, m;
vector<vector<int>> map;
vector<vector<int>> nut;

int dx[] = {0, n-1, n-1, n-1, 0, 1, 1, 1};
int dy[] = {1, 1, 0, n-1, n-1, n-1, 0, 1};


void move(int d, int v){
    --d;
    v = v % n;
    for(int i=0; i<nut.size(); ++i){
        nut[i][0] = (nut[i][0] + v*(n + dx[d])) % n;
        nut[i][1] = (nut[i][1] + v*(n + dy[d])) % n;
    }
    return;
}

void drug(){
    for(int i=0; i<nut.size(); ++i){
        int x = nut[i][0];
        int y = nut[i][1];

        ++map[x][y];
    }
}

int count(int i, int j){
    int res = 0;
    if(i > 0 && j > 0 && map[i-1][j-1] != 0) ++res;
    if(i > 0 && j < n-1 && map[i-1][j+1] != 0) ++res;
    if(i < n-1 && j > 0 && map[i+1][j-1] != 0) ++res;
    if(i < n-1 && j < n-1 && map[i+1][j+1] != 0) ++res;
    return res;
}

void addi_grow(){
    for(int i=0; i<nut.size(); ++i){
        int x = nut[i][0];
        int y = nut[i][1];

        map[x][y] += count(x, y);
        map[x][y] *= -1;
    }
    nut.clear();
}

void cut_shift(){
    for(int i=0; i<n; ++i){
        for(int j=0; j<n; ++j){
            if(map[i][j] < 0){
                map[i][j] *= -1;
                continue;
            }
            if(map[i][j] < 2) continue;
            nut.push_back({i, j});
            map[i][j] -= 2;
        }
    }
}

int total_tree(){
    int res = 0;
    for(int i=0; i<n; ++i){
        for(int j=0; j<n; ++j)
            res += map[i][j];
    }
    return res;
}

void printMap(){
    for(int i=0; i<n; ++i){
        for(int j=0; j<n; ++j){
            printf("%d ", map[i][j]);
        }printf("\n");
    }
}

void printNut(){
    for(int i=0; i<nut.size(); ++i){
        cout << nut[i][0] << ", " << nut[i][1]<<endl;
    }
}

int main() {
    cin >> n >> m;
    map = vector<vector<int>>(n, vector<int>(n));
    nut = { {n-1, 0}, {n-1, 1}, {n-2, 0}, {n-2, 1} };
    for(int i=0; i<n; ++i){
        for(int j=0; j<n; ++j){
            cin >> map[i][j];
        }
    }
    
    while(m-->0){
        int d, v;
        cin >> d >> v;
        move(d, v);
        drug();
        // printMap();
        addi_grow();
        // printMap();
        cut_shift();
        // printMap();
        // printNut();
    }
    int answer = total_tree();
    cout << answer << endl;

    map.clear();
    nut.clear();
    return 0;
}

문제 분석

  • 문제 상황
    1. n * n 격자의 게임판
    1. 특수 영양제 : 나무의 높이를 1 성장시킴 (씨앗의 경우 높이 1인 나무 생성)
      1. 특수 영양제의 초기 위치 : 격자 좌하단
      1. 이동 방향 : 1 ~ 8 ⇒ 우측 방향 (+y) 부터 반시계 방향으로 8개
      1. 이동 칸수만큼 방향대로 이동
      1. 경계를 넘어갈 경우, 반대쪽으로 넘어옴 (닫힌 공간)
    1. 1년 동안 시행
      1. 특수 영양제 입력된 규칙(방향, 칸수)대로 이동
      1. 영양제 투입 → 나무의 높이 +1
      1. 영양제 투입된 나무들은 대각선 방향 인접한 높이 1 이상의 나무 수만큼 추가 성장
      1. 투입된 나무들을 제외한 높이 2 이상의 나무들을 모두 베어내고, 그 자리에 영양제 위치
  • 목표
    1. n과 시행 횟수 (연수) m 입력
    1. 초기 나무 상태 입력
    1. m번의 시행 - 이동 규칙(방향, 칸수) 입력
    1. 시행 후 남아있는 나무의 총 길이

풀이 과정

  1. 시행의 구현 난이도는 낮았음
    1. 특수 이동, 영양제 투입, 추가 성장, 벌목 및 새로운 영양제 생성
    1. 예외 케이스가 사실상 없는 문제 상황 (시행마다 매우 독립적인 조건)
    1. 기존 영양제 위치를 제외한 벌목 대상 탐색 코드
      1. 추가 성장 때 영양제 위치의 벌목 높이에 -1 곱함으로써 표시
      1. 추가 성장의 조건이 높이 ≥ 1, 즉 높이 ≠ 0이기만 하면 되기에 사용 가능했던 방법
      1. 추가 성장 종료 직전 영양제 위치들을 저장한 벡터 clear
      1. 벌목 및 영양제 생성 때 음수의 높이는 양수로 정상화 후 continue, 2 이상의 높이는 벌목 (-2) 후 영양제 위치 벡터에 push_back
  1. runtime error : 영양제 이동 중
    1. dx, dy array를 통한 방향 벡터 정의
    1. -1로 정의된 벡터를 그대로 좌표값으로 쓸 때 음수값의 나머지도 음수로 나오는 indexing 오류
    1. n-1로 initialize 값을 수정했지만, 선언 위치에서 n값이 정의되지 않았기에 똑같은 오류 유지
    1. array 내에 -1 값을 유지하고 이동 중에 (n-1) * v(이동칸수) 로 계산
