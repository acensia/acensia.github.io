struct task{
    int t;
    int p;
    int dom;
    int idx;
    task(int a, int b, int c, int d) : t(a), p(b), dom(c), idx(d) {}
    bool operator<(const task &T) const{
        if(this->p == T.p) return this->t < T.t;
        return this->p < T.p;
    }
    bool operator>(const task &T) const{
        if(this->p == T.p) return this->t > T.t;
        return this->p > T.p;
    }
    task & operator=(const task& T){
        if(this == &T) return *this;
        
        p = T.p;
        t = T.t;
        dom = T.dom;
        idx = T.idx;
        return *this;
    }
};

vector<pair<int, int>> Judger;
priority_queue<int, vector<int>, greater<int>> JudgerPQ;

unordered_map<string, int> domainMap;

void enterWait(int t, int p, string &u){
    string domain = u.substr(0, u.find('/'));
    int ID = stoi(u.substr(u.find('/') + 1));
    if( domainMap.find(domain) == domainMap.end()){
        domainMap[domain] = domainNum++;
    }
    if( waitDomain[domainMap[domain]].find(ID) != waitDomain[domainMap[domain]].end()) return;
    waitDomain[domainMap[domain]].insert(ID);
    WQ[domainMap[domain]].push(task(t, p, domainMap[domain], ID));
    ++total;
}
