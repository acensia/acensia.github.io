    #include <iostream>

using namespace std;

int n;
vector<vector<int>> map;
vector<vector<int>> blanks;
vector<vector<int>> match;

vector<int> check_near(int i, int j, int id){
    int loved = 0;
    int blank = 0;
    if(i != 0){
        int next = map[i-1][j];
        if(next == match[id][0] || next == match[id][1] || next == match[id][2] || next == match[id][3]){
            ++loved;
        }
        else if(next == -1) ++blank;
    }
    if(i != n-1){
        int next = map[i+1][j];
        if(next == match[id][0] || next == match[id][1] || next == match[id][2] || next == match[id][3]){
            ++loved;
        }
        else if(next == -1) ++blank;
    }
    if(j != 0){
        int next = map[i][j-1];
        if(next == match[id][0] || next == match[id][1] || next == match[id][2] || next == match[id][3]){
            ++loved;
        }
        else if(next == -1) ++blank;
    }
    if(j != n-1){
        int next = map[i][j+1];
        if(next == match[id][0] || next == match[id][1] || next == match[id][2] || next == match[id][3]){
            ++loved;
        }
        else if(next == -1) ++blank;
    }
    return {loved, blank};
}

void search_seat(int id){
    int max_loved = -1;
    int max_blank = -1;
    int seat_x;
    int seat_y;
    //*
    for(int i=0; i<n; ++i){
        for(int j=0; j<n; ++j){
            if(map[i][j] != -1) continue;
            vector<int> chk = check_near(i, j, id);
            //*
            if(chk[0] > max_loved) {
                seat_x = i;
                seat_y = j;
                max_loved = chk[0];
                max_blank = chk[1];
            }
            else if(chk[0] == max_loved && chk[1] > max_blank){
                seat_x = i;
                seat_y = j;
                max_blank = chk[1];
            }//*/
            if(max_loved == 4) break;
        }
        if(max_loved == 4) break;
    }//*/
    map[seat_x][seat_y] = id;
}

  • 문제 분석
    1. 문제 상황
      1. n * n 좌석의 놀이기구
      1. n*n 명의 탑승자 with 번호
      1. 각각 좋아하는 탑승자 번호 4명씩 존재
      1. 주어진 순서대로 탑승자 및 선호 번호 정보 나열, 주어진 순서대로 탑승
      1. 탑승 규칙 (순차)
        1. 인접한 칸에 선호 번호 최다
        1. 인접한 칸에 빈칸 최다
        1. 상단, 좌측일수록
    1. 목표
      1. 주어진 순서대로 탑승 후 탑승자들의 점수 합산
      1. 점수 계산법
        1. 4명 : 1000점
        1. 3명 : 100점
        1. 2명 : 10점
        1. 1명 : 1점
        1. 0명 : 0점

풀이 과정 쟁점

  1. 구현 난이도는 어렵지 않아보임
  1. n, 놀이기구, 선호 정보, 빈칸 관리 데이터 마련
  1. 좌석 선택 알고리즘을 통한 착석은 선호 정보를 따로 저장할 필요가 없었으나, 점수계산에서 필요한 상황 발생
  1. 빈칸 관리 벡터는 빈칸 계산 용으로 쓰려 했으나 선호 탑승자 수 계산에 동시에 빈칸 계산으로 해결 → 빈칸 관리 벡터 사용하지 않음
  1. 디버깅 중 혹시나 하고 간과했던 indexing 문제 발생 (자연수 탑승자 번호 그대로 사용하며 생긴 벡터 indexing 오류) → 좌석 벡터 default값 0에서 -1로 수정
  1. 빠른 case 탐색을 위해 탈출 조건을 첫 구현 때 붙였는데, 선호 탑승자수 == 4 조건은 최우선 조건으로 사용이 맞으나 빈칸 수 == 4는 사용하면 안됐음을 확인
