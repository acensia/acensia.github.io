    #include <iostream>
#include <vector>

using namespace std;

int n, m;
vector<vector<int>> map;
vector<vector<int>> nut;

int dx[] = {0, n-1, n-1, n-1, 0, 1, 1, 1};
int dy[] = {1, 1, 0, n-1, n-1, n-1, 0, 1};


void move(int d, int v){
    --d;
    v = v % n;
    for(int i=0; i<nut.size(); ++i){
        nut[i][0] = (nut[i][0] + v*(n + dx[d])) % n;
        nut[i][1] = (nut[i][1] + v*(n + dy[d])) % n;
    }
    return;
}

void drug(){
    for(int i=0; i<nut.size(); ++i){
        int x = nut[i][0];
        int y = nut[i][1];

        ++map[x][y];
    }
}

int count(int i, int j){
    int res = 0;
    if(i > 0 && j > 0 && map[i-1][j-1] != 0) ++res;
    if(i > 0 && j < n-1 && map[i-1][j+1] != 0) ++res;
    if(i < n-1 && j > 0 && map[i+1][j-1] != 0) ++res;
    if(i < n-1 && j < n-1 && map[i+1][j+1] != 0) ++res;
    return res;
}

void addi_grow(){
    for(int i=0; i<nut.size(); ++i){
        int x = nut[i][0];
        int y = nut[i][1];

        map[x][y] += count(x, y);
        map[x][y] *= -1;
    }
    nut.clear();
}

void cut_shift(){
    for(int i=0; i<n; ++i){
        for(int j=0; j<n; ++j){
            if(map[i][j] < 0){
                map[i][j] *= -1;
                continue;
            }
            if(map[i][j] < 2) continue;
            nut.push_back({i, j});
            map[i][j] -= 2;
        }
    }
}

int total_tree(){
    int res = 0;
    for(int i=0; i<n; ++i){
        for(int j=0; j<n; ++j)
            res += map[i][j];
    }
    return res;
}

void printMap(){
    for(int i=0; i<n; ++i){
        for(int j=0; j<n; ++j){
            printf("%d ", map[i][j]);
        }printf("\n");
    }
}

void printNut(){
    for(int i=0; i<nut.size(); ++i){
        cout << nut[i][0] << ", " << nut[i][1]<<endl;
    }
}

    cin >> n >> m;
    map = vector<vector<int>>(n, vector<int>(n));
    nut = { {n-1, 0}, {n-1, 1}, {n-2, 0}, {n-2, 1} };
    for(int i=0; i<n; ++i){
        for(int j=0; j<n; ++j){

    }

    cout << answer << endl;

    map.clear();
    nut.clear();
    return 0;
}
