    #include <iostream>
#include <vector>
#include <set>

using namespace std;

int n, m;
vector<vector<int>> map;
vector<vector<int>> visit;
vector<pair<int, int>> bomb_set;
int MAX;
int total;




void add(pair<int, int> &target, pair<int, int> val){
    target.first += val.first;
    target.second += val.second;
}

void check(int i, int j, int color, pair<int, int> &coord, pair<int, int> &bomb_red, vector<pair<int, int>> &one_set){
    if(visit[i][j] == color) return;
    if(map[i][j] == -1) return;
    if(map[i][j] == -2) return;
    if(map[i][j] != 0 && map[i][j] != color) return;

    visit[i][j] = color;
    // cout << i << "-"<<j<<" for "<<color<<endl;
    if(coord.first == i && coord.second > j) coord.second = j;
    if(coord.first < i) {
        coord.first = i;
        coord.second = j;
    }
    if(map[i][j] == 0) add(bomb_red, {1, 1});
    if(map[i][j] == color) add(bomb_red, {1, 0});

    one_set.push_back({i, j});
    if(i != 0) check(i-1, j, color, coord, bomb_red, one_set);
    if(i != n-1) check(i+1, j, color, coord, bomb_red, one_set);
    if(j != 0) check(i, j-1, color, coord, bomb_red, one_set);
    if(j != n-1) check(i, j+1, color, coord, bomb_red, one_set);
}

void search(){
    visit = vector<vector<int>>(n, vector<int>(n, -1));
    pair<int, int> max_min = {1, MAX};
    pair<int, int> corner = {-1, n};

    for(int i=n-1; i>-1; --i){
        for(int j=0; j<n; ++j){
            if(visit[i][j]>0) continue;
            if(map[i][j] == -1) continue;
            if(map[i][j] == -2) continue;
            if(map[i][j] == 0) continue;
            // cout << "start"<<i<<","<<j<<endl;
            pair<int, int> bomb_red = {0, 0};
            pair<int, int> coord = {i, j};
            vector<pair<int, int>> one_set;
            int color = map[i][j];
            // start check
            check(i, j, color, coord, bomb_red, one_set);
            // cout<<bomb_red.first<<","<<bomb_red.second<<endl;
            // cout<<coord.first<<","<<coord.second<<endl;
            if(bomb_red.first == bomb_red.second) continue;
            if(bomb_red.first == 1) continue;
            if(bomb_red.first == max_min.first && bomb_red.second < max_min.second) {
                max_min.second = bomb_red.second;
                corner = coord;
                bomb_set = one_set;
            }
            if(bomb_red.first > max_min.first){
                max_min = bomb_red;
                corner = coord;
                bomb_set = one_set;
            }
        }
    }
    visit.clear();
}

void drop(){
    for(int j=0; j<n; ++j){
        int i = n-1;
        vector<int> newarr(n, -2);
        bool dropping = false;
        int it = n-1;
        while(i > -1){
            if(map[i][j] == -2) dropping = true;
            if(dropping){
                if(map[i][j] == -2){
                    --i;
                    continue;
                }
                if(map[i][j] == -1){
                    dropping = false;
                    newarr[i] = -1;
                    --i;
                    it = i;
                    continue;
                }
                else{
                    newarr[it] = map[i][j];
                    --i;
                    --it;
                }
            }
            else{
                newarr[it] = map[i][j];
                --i;
                --it;
            }
        }
        for(int i=0; i<n; ++i){
            map[i][j] = newarr[i];
        }
    }
}

void rotate(){
    vector<vector<int>> newmap(n, vector<int>(n));
    for(int i=0; i<n; ++i){
        for(int j=0; j<n; ++j){
            newmap[n-1-j][i] = map[i][j];
        }
    }
    map = newmap;
}

void print_map(){
        for(int i=0; i<n; ++i){
            for(int j=0; j<n; ++j){
                cout << map[i][j] << " ";
            }cout << endl;
        }cout <<"-----------------------------------"<<endl;
}

    // 여기에 코드를 작성해주세요.

    total = 0;
    MAX = n*n;
    map = vector<vector<int>>(n, vector<int>(n));
    for(int i=0; i<n; ++i){
        for(int j=0; j<n; ++j){
            cin >> map[i][j];
        }

    while(true){
        search();
        if(bomb_set.empty()) break;
        for(int i=0; i<bomb_set.size(); ++i){
            map[bomb_set[i].first][bomb_set[i].second] = -2;
        }
        total += bomb_set.size() * bomb_set.size();
        bomb_set.clear();
        drop();
        // print_map();
        rotate();
        // print_map();
        drop();
        // print_map();

    }

}

문제 분석

  • 문제 상황
    1. n * n 격자판
      1. -1 : 검은 돌
      1. 0 : 빨간 폭탄
      1. [1, m] : m가지 색깔의 폭탄
    1. 폭탄 묶음이 존재하는 동안 시행
      1. 폭탄 묶음 : 2개 이상으로 구성된, 빨간 색을 제외한 한 가지 색깔 or 빨간색과 한 개의 색깔로 이루어진 서로 인접한 칸들의 집합
      1. 폭탄 묶음 제거 - 우선 순위
        1. 묶음 내 폭탄 수 최다
        1. 묶음 내 빨간 폭탄 수 최소
        1. 묶음 내 기준점 (빨간 색 이외의 폭탄이 위치한 가장 하측 좌단의 칸)이 가장 하측 좌단 방향
      1. 폭탄 제거 후 모든 폭탄 중력 방향으로 낙하
        1. 중력 방향은 +x방향
        1. 검은 돌은 고정
      1. 격자판 회전 : 반시계 방향 90도
      1. 다시 폭탄 낙하
    1. 폭탄 묶음을 찾을 수 없을 시 종료
  • 목표
    1. 폭탄 제거 시 폭탄 묶음 크기의 제곱만큼 점수 추가
    1. 시행 종료 후 점수 계산

풀이 과정

  1. 행렬 내 영역 탐지
    1. 시작 color 기록, 빨간 색일 시 다른 색깔 만났을 경우 color값 변경
    1. 풀이 중 놓친 예외 케이스 발견으로 인한 다수의 변경
  1. 폭탄 낙하
    1. 열 기준 for문, 행 기준 while문
    1. 빈칸 감지 시 빈칸 통합 측정 후 폭탄 통합 측정으로 한번에 하려 시도 → 다른 방법으로 선회
    1. 대체시킬 새로운 vector 형성 후 하나의 행과 새로운 vector 동시에 iterate
      1. 폭탄 감지
        1. 낙하 중일 경우 낙하 위치에 안착, 둘다 +1
        1. 낙하 중이지 않을 경우 낙하 위치==감지 위치에 안착, 둘다 +1
      1. 빈칸 감지
        1. 낙하 중으로 변경
        1. 감지 위치만 +1
      1. 검은돌 감지
        1. 낙하 위치=감지 위치
        1. 낙하 위치==감지 위치에 검은돌 안착
        1. 둘다 +1
    1. 해당 행 새로운 vector값으로 채워넣음

쟁점

  1. 구현 난이도는 높지 않아보였으나 예외 케이스 놓침
  1. 빨간 폭탄의 변수로 인한 영역 탐지 코드의 잦은 수정
    1. visit vector로 영역 탐지 시 빨간 폭탄의 1회 탐지만 가능
    1. 빨간 폭탄부터 시작할 시 연결 가능한 색깔이 여러 개여도 하나만 감지하는 문제
    1. 빨간 폭탄부터 시작하는 case를 제거함으로 1차 해결
    1. visit을 boolean이 아닌 int로 기록해 새로운 색깔로 시작된 영역이 빨간 폭탄을 만나도 포함시킬 수 있도록 2차 해결
