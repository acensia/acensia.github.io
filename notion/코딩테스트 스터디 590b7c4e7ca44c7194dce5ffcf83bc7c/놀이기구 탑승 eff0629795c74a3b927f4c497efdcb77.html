vector<int> check_near(int i, int j, int id){
    int loved = 0;
    int blank = 0;
    if(i != 0){
        int next = map[i-1][j];
        if(next == match[id][0] || next == match[id][1] || next == match[id][2] || next == match[id][3]){
            ++loved;
        }
        else if(next == -1) ++blank;
    }
    if(i != n-1){
        int next = map[i+1][j];
        if(next == match[id][0] || next == match[id][1] || next == match[id][2] || next == match[id][3]){
            ++loved;
        }
        else if(next == -1) ++blank;
    }
    if(j != 0){
        int next = map[i][j-1];
        if(next == match[id][0] || next == match[id][1] || next == match[id][2] || next == match[id][3]){
            ++loved;
        }
        else if(next == -1) ++blank;
    }
    if(j != n-1){
        int next = map[i][j+1];
        if(next == match[id][0] || next == match[id][1] || next == match[id][2] || next == match[id][3]){
            ++loved;
        }
        else if(next == -1) ++blank;
    }
    return {loved, blank};
}

void search_seat(int id){
    int max_loved = -1;
    int max_blank = -1;
    int seat_x;
    int seat_y;

    for(int i=0; i<n; ++i){
        for(int j=0; j<n; ++j){
            if(map[i][j] != -1) continue;
            vector<int> chk = check_near(i, j, id);
            //*
            if(chk[0] > max_loved) {
                seat_x = i;
                seat_y = j;
                max_loved = chk[0];
                max_blank = chk[1];

                seat_y = j;

            }//*/
            if(max_loved == 4) break;
        }
        if(max_loved == 4) break;
    }//*/
    map[seat_x][seat_y] = id;
}
