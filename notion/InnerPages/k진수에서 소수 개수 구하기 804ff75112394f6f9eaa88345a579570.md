### 솔루션 코드

```cpp
#include <string>
#include <vector>

using namespace std;

bool if_prime(unsigned long long a){
    unsigned long long k = 2;
    if(a == 1 || a == 4) return false;
    int count = 0;
    while(k*k <= a){
        if(a%k == 0) return false;
        k++;
        /* break makes code work?? Why??
        count++;
        if(count>10000) break;//*/
    }
    return true;
}

#include <algorithm>
int solution(int n, int k) {
    int answer = 0;
    string ary = "";
    vector<string> Q;
    vector<unsigned long long> Q2;

    while(n>0){
        int c = n%k;
        
        if(c == 0) {
            if(ary != "") Q.push_back(ary);
            ary = "";
        }
        else ary += to_string(c);
        
        n/=k;
    }
    
    Q.push_back(ary);
    
    for(int i=0; i<Q.size(); ++i){
        reverse(Q[i].begin(), Q[i].end());
        unsigned long long n = stoull(Q[i]); 
        Q2.push_back(n);
    }
    
    for(int i=0; i<Q2.size(); ++i){
        if(if_prime(Q2[i])) answer++;
    }
    
    return answer;
}
```
